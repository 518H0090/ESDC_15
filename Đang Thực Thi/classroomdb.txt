CREATE DATABASE classroomdb CHARACTER SET utf8 COLLATE utf8_general_ci;
use classroomdb;

CREATE TABLE IF NOT EXISTS `users` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `username` varchar(50) NOT NULL,
 `password` varchar(50) NOT NULL,
 `fullname` varchar(255) NOT NULL,
 `birth` DATETIME NOT NULL,
 `email` varchar(255) NOT NULL,
 `phone` int NOT NULL,
 `permission`varchar(255) NOT NULL DEFAULT 'user',
 PRIMARY KEY (`id`)
 );

CREATE TABLE IF NOT EXISTS `class` (
 `id` int NOT NULL AUTO_INCREMENT,   
 `code` text NOT NULL UNIQUE,
 `classname` varchar(50) NOT NULL,
 `description` text NOT NULL,
 `room` varchar(50) NOT NULL,
 `image_class` longblob NOT NULL,
 `instructor` varchar(255) NOT NULL,
 `own` varchar(255) NOT NULL,
 PRIMARY KEY (`id`)
 );

CREATE TABLE IF NOT EXISTS `attend` (
 `id` int NOT NULL AUTO_INCREMENT,   
 `code` varchar(255) NOT NULL,
 `username` varchar(50) NOT NULL,
 `permission` varchar(255) NOT NULL DEFAULT 'student',
 `fullname` varchar(50) NOT NULL,
 PRIMARY KEY (`id`)
 );

CREATE TABLE IF NOT EXISTS `post` (
 `id` int NOT NULL AUTO_INCREMENT,   
 `code` varchar(255) NOT NULL,
 `POSTD` varchar(255) NOT NULL UNIQUE,
 `username` varchar(50) NOT NULL,
 `message` TEXT NOT NULL,
 `file` longblob NOT NULL,
 PRIMARY KEY (`id`)
 );

CREATE TABLE IF NOT EXISTS `comment` (
 `id` int NOT NULL AUTO_INCREMENT,   
 `code` varchar(255) NOT NULL,
 `POSTD` varchar(255) NOT NULL,
 `username` varchar(50) NOT NULL,
 `message` TEXT NOT NULL,
 PRIMARY KEY (`id`)
 );
